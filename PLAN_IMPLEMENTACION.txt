--- PASO 2: Migracion de roles y estructura de locales ---

PROBLEMA ACTUAL:
  - Roles: solo ADMIN y LOCAL
  - usuarios.local_id = un solo local por usuario (no sirve para Fiambrisima)

SOLUCION:
  A) Cambiar el ENUM de roles en MySQL:
     ADMIN, LOCAL  ->  ADMIN, ENCARGADO, REPOSITOR

  B) Crear tabla usuario_locales (relacion muchos a muchos):
     usuario_locales (id, usuario_id, local_id)
     - Fiambrisima: 1 usuario vinculado a sus 4 locales
     - Repositores: 1 repositor vinculado a las tiendas que le asigne el admin

  C) Migrar los usuarios LOCAL existentes a ENCARGADO

  D) Eliminar columna local_id de la tabla usuarios

  SQL A CORRER (Henry lo ejecuta en MySQL antes de continuar):

    ALTER TABLE usuarios MODIFY COLUMN rol ENUM('ADMIN','ENCARGADO','REPOSITOR') NOT NULL DEFAULT 'ENCARGADO';

    CREATE TABLE IF NOT EXISTS usuario_locales (
      id INT AUTO_INCREMENT PRIMARY KEY,
      usuario_id INT NOT NULL,
      local_id INT NOT NULL,
      UNIQUE KEY uq_usuario_local (usuario_id, local_id),
      FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
      FOREIGN KEY (local_id) REFERENCES locales(id) ON DELETE CASCADE
    );

    INSERT INTO usuario_locales (usuario_id, local_id)
    SELECT id, local_id FROM usuarios WHERE local_id IS NOT NULL AND rol = 'LOCAL';

    UPDATE usuarios SET rol = 'ENCARGADO' WHERE rol = 'LOCAL';

    ALTER TABLE usuarios DROP COLUMN local_id;

    CREATE TABLE IF NOT EXISTS activacion_ejecucion (
      id INT AUTO_INCREMENT PRIMARY KEY,
      activacion_id INT NOT NULL,
      local_id INT NOT NULL,
      repositor_id INT NOT NULL,
      ejecutada TINYINT DEFAULT 0,
      fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY (activacion_id) REFERENCES activaciones(id),
      FOREIGN KEY (local_id) REFERENCES locales(id),
      FOREIGN KEY (repositor_id) REFERENCES usuarios(id)
    );

    CREATE TABLE IF NOT EXISTS reclamos (
      id INT AUTO_INCREMENT PRIMARY KEY,
      repositor_id INT NOT NULL,
      local_id INT NOT NULL,
      mensaje TEXT NOT NULL,
      foto VARCHAR(255),
      leido TINYINT DEFAULT 0,
      creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY (repositor_id) REFERENCES usuarios(id),
      FOREIGN KEY (local_id) REFERENCES locales(id)
    );

      unidad_medida ENUM('FARDO','CORTE','PALE','UNIDAD') NOT NULL,

     ENCARGADO puede:
     - Ir a "Cargar pedido"
     - Elegir fecha de carga y fecha de entrega (1-2 dias despues)
     - Ver tabla de productos con codigo, precio, cantidad
     - Ingresar cantidades (respetando la unidad: fardo/corte/pale)

     - Enviar pedido -> va al admin

     - Ver todos los pedidos
     - Confirmar recepcion -> el local ve el ticket con fechas
     - Rechazar con comentario

  Tablas necesarias:

    CREATE TABLE pedidos (
      id INT AUTO_INCREMENT PRIMARY KEY,
      local_id INT NOT NULL,
      usuario_id INT NOT NULL,
      fecha_carga DATE NOT NULL,
      fecha_entrega DATE NOT NULL,
      estado ENUM('PENDIENTE','CONFIRMADO','RECHAZADO') DEFAULT 'PENDIENTE',
      comentario_rechazo TEXT,
      creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY (local_id) REFERENCES locales(id),
      FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
    );

    CREATE TABLE pedido_items (
      id INT AUTO_INCREMENT PRIMARY KEY,
      pedido_id INT NOT NULL,
      producto_id INT NOT NULL,
      cantidad DECIMAL(10,2) NOT NULL,
      unidad VARCHAR(50),
      FOREIGN KEY (pedido_id) REFERENCES pedidos(id) ON DELETE CASCADE,
      FOREIGN KEY (producto_id) REFERENCES productos(id)
    );
